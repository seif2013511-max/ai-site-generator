async function generateSite() { const prompt = document.getElementById('userPrompt').value.trim(); const btn = document.getElementById('genBtn'); const iframe = document.getElementById('preview'); if (!prompt) { alert("اكتب وصف الأول يا سيف!"); return; } btn.disabled = true; btn.innerHTML = "جاري البناء... 🏗️"; try { const response = await fetch('/api/generate', { method: 'POST', headers: { 'Content-Type': 'application/json' }, body: JSON.stringify({ prompt: prompt }) }); const data = await response.json(); if (response.ok && data.code) { iframe.srcdoc = data.code; } else { alert("خطأ من السيرفر: " + (data.error || "مشكلة في المفتاح")); } } catch (error) { console.error(error); alert("فشل الاتصال: تأكد إنك ضفت API_KEY في Vercel"); } finally { btn.disabled = false; btn.innerHTML = "توليد الموقع الآن ✨"; } }
